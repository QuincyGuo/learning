    # pay组件的基本数据及页面

    不允许给生命周期函数加async。

    1、mounted发送请求，获取数据
  async mounted () {
    let result = await this.$API.reqGetPayInfo(tradeNo, data);	// 向服务器发送请求
  },
// 本来这里是这样写，异步向服务器发送请求，获取数据；但是实际上尽量不要给生命周期函数前加async，所以。把这里本来要写的异步代码封装成一个函数，放到methods里。
  
2、实际的写法
  mounted () {
    // mounted调用getPayInfo，向服务器发送请求，获取订单支付数据
    this.getPayInfo();
  },

  methods: {
    // 向服务器发送请求，获取订单支付数据
    async getPayInfo () {
      // 1、发送请求
      let result = await this.$API.reqGetPayInfo(this.orderId);
      // 2、获取服务器返回结果并进行处理
      if (result.code == '200') {
        // 2.1、请求成功，则获取服务器传回的支付信息数据，存储到组件当中
        this.payInfo = result.data;
      } else {
        // 2.2、请求失败则打印失败原因
        console.log(result.message);
      }
    }
  },

    this.payInfo里的微信支付二维码数据目前是一段字符串，将其生成二维码并展示

    # qrcode插件

    1、安装：
npm i qrcode --save 
2、引入并使用
import QRCode from 'qrcode'	// 引入QRCode对象，里面包含一些函数方法
let result = QRCode.toDataURL(this.payInfo.codeUrl);	// QRCode.toDataURL，将二维码字符串转变为图片的方法
其返回结果是一个promise，所以写成异步函数，等待其成功后的返回结果就是生成的图片地址

}

    # 查询支付状态

    定时向服务器发送查询支付状态的请求，成功则关闭定时器，关闭弹窗，跳转至下一路由。这里在组件中新建两个data，time:null, code:'',

    // 如果还没有定时器则开启一个定时器
    if (!this.timer) {
    this.timer = setInterval(async () => {
      // 定时向服务器发送查询支付状态的请求
      let result = this.$API.reqPayStatus(this.orderId);
      // 如果已支付成功
      if (result.code == 200) {
        // 清除定时器并将this.timer置空
        clearInterval(this.timer);
        this.timer = null;
        // 保存返回的支付状态码
        this.code = result.code;
        // 关闭弹窗，并跳到下一路由
        this.$msgbox.close();
        this.$router.push('/paysuccess');
      }
    }, 2000);
}

    # 点击已完成支付和支付遇到问题

    beforeClose是MessageBox组件的一个属性，是点击弹窗按钮的方法的所在位置

beforeClose: (type, instance, done) => {
  //type：区分取消|确定按钮
  //instance：当前组件实例
  //done：关闭弹出框的方法

  if (type == 'cancel') {
    // 点击支付有问题：弹出提醒消息，清除定时器并将this.timer置空，关闭弹窗
    alert('请联系管理人员');
    clearInterval(this.timer);
    this.timer = null;
    done();
  }
  else {
    // 点击支付成功：确认是否真的已支付成功，清除定时器并将this.timer置空，关闭弹窗，并跳到下一路由
    if (this.code == 200) {
      // 这里的this.code是前面弹窗出现后，定时向服务器发送的查询支付状态返回的结果保存到本组件的data
      clearInterval(this.timer);
      this.timer = null;
      this.$msgbox.close();
      this.$router.push('/paysuccess');
    }
  }
}
