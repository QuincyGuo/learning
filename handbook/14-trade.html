    # 功能2-点击切换默认地址

    1、添加默认地址为computed属性
computed: {
    // 默认地址
    userDefaultAddress () {
      // 返回用户地址数组中，默认属性为1的地址
      return this.userAddress.find(item => item.isDefault == 1);
    }
},
2、给html标签添加事件
<p @click="changeDefault(address,userAddress)">
   <span>
   。。。
</p>
3、书写事件
methods: {
    // 修改默认地址
    changeDefault (address, userAddress) {
      userAddress.forEach((element) => {
        element.isDefault = 0;
      });
      address.isDefault = 1;
    }
},

    # 提交订单

    向服务器提交订单，发送提交订单的请求

    # api

    // 提交订单
// URL:/api/order/auth/submitOrder?tradeNo={tradeNo}  mehtod:post   需要带参数tradeNo, data{consigne,totalPrice...}
export const reqSubmitOrder = (tradeNo, data) =>
  requests({
    url: `order/auth/submitOrder?tradeNo=${tradeNo}`,
    data,
    method: "post",
  });

    # 从这往后，就不再使用vuex，而是将数据就存在组件本身里

    # 集中引入api

    1、以前，要在组件页面中使用api，需要在每个组件里import { reqAddressInfo, reqOrderInfo } from "../../api";
2、引入
现在，main.js里
// 统一引入api接口
import * as API from "@/api";
new Vue({
  。。。
  beforeCreate() {
    Vue.prototype.$API = API; // 将所有的api挂载在vue原型对象上，以后就不再需要在组件里一个个引入
  },
}).$mount("#app");
3、使用
this.$API.reqSubmitOrder(tradeNo, data);	// 向服务器发送请求

    # 添加事件并提交订单

    // 这里的函数注释，可以作为自己以后整理的范例

    <a class="subBtn" @click="submitOrder">提交订单</a>
methods: {
    async submitOrder () {
      // 1、准备好需要发送请求的参数
      let tradeNo = this.userOrderInfo.tradeNo;
      let data = {
        consignee: this.userDefaultAddress.consignee,
        consigneeTel: this.userDefaultAddress.phoneNum,
        deliveryAddress: this.userDefaultAddress.fullAddress,
        paymentWay: "ONLINE",
        orderComment: this.msg,
        orderDetailList: this.userOrderInfo.detailArrayList,    // 最终要购买的商品数据数组，数组每项的内容是商品名称、价格等数据
      }
      // 2、发送请求
      let result = await this.$API.reqSubmitOrder(tradeNo, data);
      // 3、获取服务器返回结果并进行处理
      if (result.code == '200') {
        // 3.1、请求成功，则获取服务器传回的订单号，并进行路由跳转（且携带订单号作为参数进行跳转）
        this.orderId = result.data;
        this.$router.push('/pay?orderId=' + this.orderId);
      } else {
        ///3.2、请求失败则打印失败原因
        console.log(result.message);
      }
    }
},
